CREATE TABLE CARS
(
    ID int Primary Key NOT NULl AUTO_INCREMENT,
    RegPlate varchar(32) UNIQUE not null, 
    Maker varchar(32) not null,
    Model varchar(32) not null,
    ManufacturedYear int not null,
    Engine int not null,
    Type int not null,
    BodyType varchar(32) not null,
    FuelConsumption double not null,
	Image varchar(64) not null
	CONSTRAINT BodyFK FOREIGN KEY (Type) 
                     REFERENCES Car_Type(ID),
);

CREATE TABLE Car_Type
(
    ID int Primary Key NOT NULl AUTO_INCREMENT,
    Type varchar(32) UNIQUE not null, 
    Price double not null,
);

CREATE TABLE CLIENTS
(
    ID int Primary Key NOT NULl AUTO_INCREMENT,
    Pesel numeric(11) not null UNIQUE,
    Name varchar(32) not null,
    Surname varchar(32) not null,
    Born date not null,
    IsMale boolean not null,
    PhoneNumber int not null,
    Address varchar(32) not null,
    City varchar(32) not null,
    Type varchar(32) not null,
    Image varchar(64) not null
);

CREATE TABLE RENTS
(
    ID int Primary Key NOT NULl AUTO_INCREMENT,
    CarID int not null,
    CONSTRAINT CarFK FOREIGN KEY (CarID) 
                     REFERENCES CARS(ID),
    ClientID int not null,
    Constraint ClientFK FOREIGN KEY (ClientID) 
                     REFERENCES CLIENTS(ID),
    RentStart date not null,
    RentEnd date not null,
	isReturned boolean not null
);

CREATE TABLE car_bodytype (
  bodytype varchar(32) NOT NULL UNIQUE
);

CREATE TABLE car_brand
(
	brand varchar(32) NOT NULL UNIQUE
);
